%%% Macro shorthand to add figures
%%%

\newcommand{\psyfig}[3]{%
    \begin{figure}[h!]
    \centering
    \includegraphics[max width=\linewidth, keepaspectratio]{#1}
    \caption{#2}
    \label{fig:#3}
    \end{figure}}
