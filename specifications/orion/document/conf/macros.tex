% For quick insertion of code snippets
\newcommand{\pcoded}[4]{\lstinputlisting[label=#1,caption=#2,language=#3]{#4}}
\newcommand{\pcode}[3]{\lstinputlisting[label=#1,caption=#2,language=ruby]{#3}}

% global ref to document link

% For printing output
\newcommand{\poutput}[3]{\lstinputlisting[label=#1,caption=#2,language=]{#3}}

\newcommand{\psybox}[1]{
\fcolorbox{psyan}{lightpsyan}{
  \parbox{0.93\textwidth}{
    #1
  }}}

% Quick way for adding figures
\newcommand{\psyfig}[4]{
  \begin{figure}[hb]
  \centering
  \includegraphics[width=#4]{#1}
  \caption{#2}
  \label{#3}
  \end{figure}}

\newcommand{\itemit}[1]{\item\textit{#1}}
\newcommand{\itembf}[1]{\item\textbf{#1}}
