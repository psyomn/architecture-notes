% A few reusable macros.

% For quick insertion of code snippets
\newcommand{\pcoded}[4]{\lstinputlisting[label=#1,caption=#2,language=#3]{#4}}
\newcommand{\haskellcode}[3]{\lstinputlisting[label=#1,caption=#2,language=haskell]{#3}}

% For printing output
\newcommand{\poutput}[3]{\lstinputlisting[label=#1,caption=#2,language=]{#3}}

\newcommand{\psybox}[1]{
\fcolorbox{psyan}{lightpsyan}{
  \parbox{0.93\textwidth}{
    #1
  }}}

%% Note width as inches
\newcommand{\psyfig}[4]{
  \begin{figure}[hb]
  \centering
  \includegraphics[width=#4]{#3}
  \caption{#2}
  \label{#1}
  \end{figure}}
